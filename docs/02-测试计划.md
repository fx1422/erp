## 1. 测试策略

### 1.1 测试目标

确保 ERP 系统 MVP 版本满足产品需求文档中定义的功能要求，保证系统：
- 功能完整性和正确性
- 数据一致性和完整性
- 系统安全性和稳定性
- 用户体验符合预期

### 1.2 测试原则

| 原则 | 说明 |
|------|------|
| 尽早测试 | 测试活动与开发并行，左移测试 |
| 分层测试 | 单元测试 → 集成测试 → 系统测试 → 验收测试 |
| 自动化优先 | 核心业务流程和回归测试自动化 |
| 风险驱动 | 优先测试高风险、高优先级功能 |
| 可追溯性 | 测试用例与需求双向追溯 |

### 1.3 测试层次

```
┌─────────────────────────────────────────────────────────┐
│                    验收测试 (UAT)                         │
│              业务场景验证、用户验收测试                     │
└─────────────────────────────────────────────────────────┘
                          ↑
┌─────────────────────────────────────────────────────────┐
│                    系统测试 (E2E)                         │
│           端到端测试、UI 自动化、性能测试                    │
└─────────────────────────────────────────────────────────┘
                          ↑
┌─────────────────────────────────────────────────────────┐
│                    集成测试                               │
│           模块间交互、API 集成、数据流验证                   │
└─────────────────────────────────────────────────────────┘
                          ↑
┌─────────────────────────────────────────────────────────┐
│                    单元测试                               │
│           函数/方法级测试、代码覆盖率                      │
└─────────────────────────────────────────────────────────┘
```

### 1.4 测试类型

| 测试类型 | 工具 | 覆盖率目标 | 执行阶段 |
|----------|------|------------|----------|
| 单元测试 | Vitest / Jest | ≥80% | 开发阶段 |
| 接口测试 | Postman / Supertest | 100% API | 集成阶段 |
| E2E 测试 | Playwright / Cypress | 核心流程 100% | 系统测试 |
| 性能测试 | k6 / Artillery | 关键接口 | 发布前 |
| 安全测试 | OWASP ZAP | 高危漏洞 0 | 发布前 |

import { UserFactory } from '@/test/factories/user.factory';
import { hashPassword } from '@/utils/crypto';

describe('UserService', () => {
  let userService: UserService;
  let mockPrisma: Partial<PrismaClient>;

    mockPrisma = {

        findUnique: vi.fn(),
        update: vi.fn(),

### 5.5 CI/CD 集成

```yaml
# .github/workflows/test.yml
name: Test Pipeline

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main, develop]

jobs:
  backend-test:
    runs-on: ubuntu-latest
    services:
      postgres:
        image: postgres:15
        env:
          POSTGRES_PASSWORD: postgres
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5
        ports:
          - 5432:5432
      redis:
        image: redis:7-alpine
        ports:
          - 6379:6379

    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
          cache-dependency-path: erp-backend/package-lock.json

      - name: Install dependencies
        working-directory: ./erp-backend
        run: npm ci

      - name: Run database migrations
        working-directory: ./erp-backend
        run: npx prisma migrate deploy
        env:
          DATABASE_URL: postgresql://postgres:postgres@localhost:5432/postgres

      - name: Run unit tests
        working-directory: ./erp-backend
        run: npm run test:unit
        env:
          DATABASE_URL: postgresql://postgres:postgres@localhost:5432/postgres
          REDIS_URL: redis://localhost:6379

      - name: Run integration tests
        working-directory: ./erp-backend
        run: npm run test:integration
        env:
          DATABASE_URL: postgresql://postgres:postgres@localhost:5432/postgres
          REDIS_URL: redis://localhost:6379

      - name: Upload coverage
        uses: codecov/codecov-action@v4
        with:
          files: ./erp-backend/coverage/coverage-final.json
          flags: backend

  frontend-test:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
          cache-dependency-path: erp-frontend/package-lock.json

      - name: Install dependencies
        working-directory: ./erp-frontend
        run: npm ci

      - name: Run unit tests
        working-directory: ./erp-frontend
        run: npm run test:unit

      - name: Run E2E tests
        working-directory: ./erp-frontend
        run: npm run test:e2e

      - name: Upload E2E report
        uses: actions/upload-artifact@v4
        if: always()
        with:
          name: playwright-report
          path: erp-frontend/playwright-report/
          retention-days: 30

  quality-gates:
    runs-on: ubuntu-latest
    needs: [backend-test, frontend-test]

    steps:
      - name: Check coverage thresholds
        run: |
          echo "Backend coverage: $(cat erp-backend/coverage/coverage-summary.json | jq '.total.lines.pct')"
          echo "Frontend coverage: $(cat erp-frontend/coverage/coverage-summary.json | jq '.total.lines.pct')"
          
          # 检查是否达到 80% 覆盖率
          BACKEND_COV=$(cat erp-backend/coverage/coverage-summary.json | jq '.total.lines.pct')
          FRONTEND_COV=$(cat erp-frontend/coverage/coverage-summary.json | jq '.total.lines.pct')
          
          if (( $(echo "$BACKEND_COV < 80" | bc -l) )); then
            echo "Backend coverage below 80%"
            exit 1
          fi
          
          if (( $(echo "$FRONTEND_COV < 80" | bc -l) )); then
            echo "Frontend coverage below 80%"
            exit 1
          fi
```

## 7. 缺陷管理

### 7.1 缺陷等级定义

| 等级 | 名称 | 定义 | 响应时间 | 修复时限 |
|------|------|------|----------|----------|
| P0 | 致命 | 系统崩溃、数据丢失、核心功能不可用 | 2 小时 | 24 小时 |
| P1 | 严重 | 主要功能失效、数据错误 | 4 小时 | 48 小时 |
| P2 | 一般 | 次要功能问题、UI 问题 | 24 小时 | 1 周 |
| P3 | 轻微 | 建议性问题、体验优化 | 48 小时 | 下一版本 |
